Правильная логика:
// Мультиграф = есть два или более ребра, соединяющих одну и ту же пару вершин
const edgeSignature = (row: number[]) => {
  const vertices = row.map((val, idx) => Math.abs(val) > 0 ? idx : -1).filter(v => v !== -1).sort((a,b) => a-b);
  return vertices.join(','); // "0,2" для ребра между вершинами 0 и 2
};
const signatures = data.map(edgeSignature);
